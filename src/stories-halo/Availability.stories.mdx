export function AvailabilityTableData(token) {
  return (
     <tr key={token.name}>
          <td>
              {token.name}
          </td>
          <td>
              {token.version}
          </td>
          <td>
              {token.status}
          </td>
      </tr>
  )
}

export const VersionBase = ({packages}) => {
  let tableRows = packages.map(packageItem => AvailabilityTableData(packageItem));
  return tableRows;
};
