export function MercerAvailabilityTableData(token) {
  return (
     <tr key={token.name}>
          <td>
              {token.name}
          </td>
          <td>
              {token.version}
          </td>
          <td>
              {token.status}
          </td>
      </tr>
  )
}

export const VersionBase = ({packages}) => {
  let tableRows = packages.map(packageItem => MercerAvailabilityTableData(packageItem));
  return tableRows;
};
